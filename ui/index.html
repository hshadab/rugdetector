x402 Protocol (JavaScript)
// Create X-PAYMENT header
const paymentPayload = {
  scheme: 'exact',
  network: 'base',
  currency: 'USDC',
  amount: '0.1',
  txHash: '0xYOUR_TX_HASH'
};

const xPaymentHeader = btoa(JSON.stringify(paymentPayload));

// Make request with X-PAYMENT header
const response = await fetch('https://rugdetector.ai/check', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-PAYMENT': xPaymentHeader
  },
  body: JSON.stringify({
    contract_address: '0x5C69bEe701ef814a2B6a3EDD4B1652CB9cc5aA6f',
    blockchain: 'ethereum'
  })
});

// Handle 402 Payment Required
if (response.status === 402) {
  const xPaymentResponse = response.headers.get('X-PAYMENT-RESPONSE');
  const paymentDetails = JSON.parse(atob(xPaymentResponse));
  console.log('Payment required:', paymentDetails);
} else {
  const data = await response.json();
  console.log('Analysis:', data);
}
